start : OPCLEAR AL     #AL=0
        OPCLEAR CL     #CL=0
        OPLOADVB CL 1  #CL=1
        OPLOADVB AL 1  #AL=1     将值存入寄存器中
add:    OPLOAD AL CL   #AL=CL
        OPADDB AL 10   #AL=AL+[10]
        OPSTOREB 10 AL #[10]=AL  将寄存器内容加载到内存
        OPINC CL       #CL=CL+1
        OPCMPVB CL 4   #CMP 4
        OPJG    quit
        OPJNG    add
quit:   OPPRINTR AL
        OPHALT
